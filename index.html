3. Reverse the number

            const num = 15463;
            function reverseNumber(num) {
                const str = num.toString();
                let reversedNumber = "";
                for (let i = str.length - 1; i >= 0; i--) {
                  reversedNumber += str.charAt(i);
                }
                return parseInt(reversedNumber);
              }
            reverseNumber(num);
